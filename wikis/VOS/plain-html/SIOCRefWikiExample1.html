Query

PREFIX  rdf:    <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX  sioc:   <http://rdfs.org/sioc/ns#>
PREFIX  sioct:  <http://rdfs.org/sioc/types#>
SELECT DISTINCT ?attribute
WHERE {
        ?forum  rdf:type           sioct:Wiki  .
        ?forum  sioc:container_of  ?post       .
        ?post   ?attribute         ?o  
      }
ORDER BY ?attribute
