Dump all Provinces


PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX ndw:<http://demo.openlinksw.com/schemas/northwind#>
SELECT ?name ?country
WHERE
  { 
    ?sp rdf:type ndw:Province.
    ?sp ndw:provinceName ?name.
    ?sp ndw:is_province_of ?ipo .
    ?ipo ndw:name ?country
  }
