Dump of phpBB Groups and their members


PREFIX sioc: <http://rdfs.org/sioc/ns#>
SELECT ?grp ?username
WHERE 
  {
    ?group a sioc:Usergroup .
    optional{ ?group sioc:name ?grp }.      
    optional{ ?group sioc:has_member ?member} .
    optional{ ?member sioc:name ?username} .
  }
ORDER BY DESC (?grp)
