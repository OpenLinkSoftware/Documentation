PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
SELECT *
WHERE 
  {
    ?forum a sioct:Briefcase .
    ?forum sioc:container_of ?post.
    optional { ?post sioc:topic ?topic } .
    optional { ?topic rdf:type skos:Concept } .
    optional { ?topic skos:prefLabel ?tag }
  }
